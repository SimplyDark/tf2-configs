event MenuOpen
{
    StopEvent   MenuClose 0.0 

    // fade in
    Animate HudMenu Alpha       "255"       Linear 0.0 0.0
    Animate HudMenu SelectionAlpha  "255"       Linear 0.0 0.0
    Animate HudMenu FgColor     "FgColor"       Linear 0.0 0.0
